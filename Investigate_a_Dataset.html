def money_relevance(column,step):
    '''column: is for column input.
       step: is for the number of points each xtick is apart.
    '''
    avr=df.groupby(str(column)).sum().reset_index()
    avr=pd.DataFrame(avr)
    plt.figure(figsize=(10,10))
    col=[ 'budget', 'revenue',  'budget_adj', 'revenue_adj' ]
    for z in col:
        plt.plot(avr[column],avr[z])
        plt.xticks(avr[column][::step])


    plt.legend(col,fontsize=14)
    
def top(column,step):
